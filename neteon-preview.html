Simple Integration

Production-ready in minutes with our intuitive API

// Initialize Neteon SDK
import { Neteon } from '@neteon/sdk';

const client = new Neteon({
  apiKey: process.env.NETEON_API_KEY,
  region: 'us-west-2'
});

// Deploy your model to the edge
const deployment = await client.deploy({
  model: 'my-ai-model',
  target: 'edge',
  replicas: 3,
  autoScale: true
});

console.log(`Deployed: ${deployment.url}`);
